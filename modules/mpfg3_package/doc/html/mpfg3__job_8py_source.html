230  def get_catalog_from_image(self, img_name, cat_extension=".txt"):
231  """!
232  Convenience method to find the catalog name matching an image name, for instance
233  "starfield_image" -> "star_catalog", or "image.fits" -> "galaxy_catalog"
234  @param img_name name or image file type of an image
235  @param cat_extension expected file extension for the catalog (default: .txt)
236  @return a catalog name among:
237  - "deep_star_catalog"
238  - "star_catalog"
239  - "deep_galaxy_catalog"
240  - "galaxy_catalog"
241  @see see get_file_types. __init__
242  @note: the same catalog filename may have more than one extension (like .txt and .fits)
243  """
244  cat_name = None
245 
246  if "deep_starfield_image" in img_name:
247  cat_name = "deep_star_catalog" # a deep star catalog
248  elif "starfield_image" in img_name:
249  cat_name = "star_catalog" # a star catalog
250  elif "deep_image" in img_name:
251  cat_name = "deep_galaxy_catalog" # a deep galaxy catalog
252  elif "image" in img_name:
253  cat_name = "galaxy_catalog" # a galaxy catalog
254 
255  if not cat_name is None:
256  cat_name += cat_extension
257 
258  return cat_name

289  def get_stamp_size(self):
290  """!
291  Return the image postage stamp size based on the branch
292  @return postage stamp size for the job's branch
293  """
294  #print "branch:", self.branch, "ground in branch:", "ground" in self.branch
295 
296  if self.branch != "multiepoch" and self.branch != "full":
297  if "ground" in self.branch:
298  return 48
299  elif "space" in self.branch:
300  return 96
301  else:
302  return 48
303 
304  elif self.branch == "multiepoch":
305  # Assumes G3 space images are coadded with size 96
306  if "ground" in self.branch:
307  return 48
308  elif "space" in self.branch:
309  return 96
310  else:
311  return 48
312 
313  elif self.branch == "full":
314  if "ground" in self.branch:
315  return 48
316  elif "space" in self.branch:
317  return 96
318  else:
319  return 48

02186                                                    "PRIMARY_DATASET.CATALOG_PROPERTIES.GALAXY")

02189       galaxy_catalog = self._open_catalog(self.galaxy_catalog_filepath, 
02190                                           self.galaxy_catalog_is_sextractor,
02191                                           self.galaxy_catalog_hdu_no, job, worker)
02192 
02193       # --- Available SExtractor column names
02194       cat_colnames = galaxy_catalog.get_col_names()   
02195 
02196 
02197       # --- Coordinates of galaxies, to be converted to coordimates of stamps lower-left corners
02198       xc_colname = col_mapping_dico.get("GAL_x", None)    # x coordinate in mosaic
02199       yc_colname = col_mapping_dico.get("GAL_y", None)    # y coordinate in mosaic
02200       Xc_colname = col_mapping_dico.get("GAL_X", None)    # Physical X centroid
02201       Yc_colname = col_mapping_dico.get("GAL_Y", None)    # Physical Y centroid
02202 
02203       if xc_colname is None or yc_colname is None:
02204          msg = "{0} - {1} and/or {2} missing from the 'COL_MAPPING' key".format(
02205                                                   worker.name, "GAL_x", "GAL_y")     
02206          raise GfitShapeEstimator.ShapeMeasurementError(msg)    
02207 
02208       if Xc_colname is None or Yc_colname is None:
02209          msg = "{0} - {1} and/or {2} missing from the 'COL_MAPPING' key".format(
02210                                                   worker.name, "GAL_X", "GAL_Y")     
02211          raise GfitShapeEstimator.ShapeMeasurementError(msg)    
02212 
02213       # --- Mapped coordinate columns are required
02214       if not xc_colname in cat_colnames or not yc_colname in cat_colnames:
02215          msg = "{0} - The {1} and {2} columns must be available in the galaxy catalog "\
02216                "for processing to continue".format(worker.name, xc_colname, yc_colname)     
02217          raise GfitShapeEstimator.ShapeMeasurementError(msg)
02218          
02219       if not Xc_colname in cat_colnames or not Yc_colname in cat_colnames:
02220          msg = "{0} - The {1} and {2} columns must be available in the galaxy catalog "\
02221                "for processing to continue".format(worker.name, Xc_colname, Yc_colname)     
02222          raise GfitShapeEstimator.ShapeMeasurementError(msg)         
02223 
02224       xc_array = galaxy_catalog.get_named_col_data(xc_colname)
02225       yc_array = galaxy_catalog.get_named_col_data(yc_colname)
02226 
02227       Xc_array = galaxy_catalog.get_named_col_data(Xc_colname)
02228       Yc_array = galaxy_catalog.get_named_col_data(Yc_colname)
02229 
02230       # --- Force conversion to stamp lower-left corner coordinates
02231       x_array = numpy.floor(xc_array / self.galaxy_stamp_size) * self.galaxy_stamp_size
02232       y_array = numpy.floor(yc_array / self.galaxy_stamp_size) * self.galaxy_stamp_size
02233 
02234       # --- Record galaxy coordinates
02235       galaxy_x_y_coords   = zip(x_array, y_array)
02236       galaxy_xc_yc_coords = zip(xc_array, yc_array)
02237       galaxy_Xc_Yc_coords = zip(Xc_array, Yc_array)
02238 
02239       info_dico["GAL_xc_yc"] = dict(zip(galaxy_x_y_coords, galaxy_xc_yc_coords))
02240       info_dico["GAL_Xc_Yc"] = dict(zip(galaxy_x_y_coords, galaxy_Xc_Yc_coords))
02241 
02242       # --- Object ID (or sequence number)
02243       id_colname = col_mapping_dico.get("GAL_id", None)  # object id
02244       if id_colname is not None:
02245          if not id_colname in cat_colnames:
02246             msg = "{0} - The {1} column must be available in the galaxy catalog "\
02247                   "for processing to continue".format(worker.name, id_colname)     
02248             raise GfitShapeEstimator.ShapeMeasurementError(msg)         
02249          id_array = galaxy_catalog.get_named_col_data(id_colname)
02250       else:
02251          id_array = numpy.arange(0, len(galaxy_catalog.get_col_data(xc_colname)))
02252    
02253       info_dico["GAL_id"] = dict(zip(galaxy_x_y_coords, id_array))
02254 
02255       # --- Object RA and DEC if available
02256       RA_colname  = col_mapping_dico.get("GAL_RA",  None)    # RA  physical coordinates
02257       DEC_colname = col_mapping_dico.get("GAL_DEC", None)    # DEC physical coordinates
02258       if RA_colname is not None or DEC_colname is not None:
02259          if not RA_colname in cat_colnames or not RA_colname in cat_colnames:
02260             msg = "{0} - The {1} or the {2} column is not available in the galaxy catalog - "\
02261                   "Ignored".format(worker.name, RA_colname, DEC_colname)     
02262          else:
02263             RA_array  = galaxy_catalog.get_named_col_data(RA_colname)
02264             DEC_array = galaxy_catalog.get_named_col_data(DEC_colname)
02265             galaxy_RA_DEC_coords = zip(RA_array, DEC_array)
02266             info_dico["GAL_RA_DEC"] = dict(zip(galaxy_x_y_coords, galaxy_RA_DEC_coords))
02267 
02268       required_colnames = info_dico.keys()
02269       excluded_columns = ["GAL_x", "GAL_y", "GAL_X", "GAL_Y", "GAL_RA", "GAL_DEC"]
02270       excluded_columns.extend(required_colnames)
02271 
02272       # --- Keep track of extra SE columns if specified
02273       for colname in col_mapping_dico.keys():
02274          if not colname in excluded_columns:
02275             mapped_col_name = col_mapping_dico[colname]
02276             if mapped_col_name in cat_colnames:
02277                coldata = galaxy_catalog.get_named_col_data(mapped_col_name)
02278                info_dico[colname] = dict(zip(galaxy_x_y_coords, coldata))
02279 
02280       return info_dico

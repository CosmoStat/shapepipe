485  def _parse_config(self):
486  """!
487  Parse each configuration line and create appropriate data structures
488  """
489 
490  with open(self._input_config_path) as fd:
491 
492  section_stack = [] # to keep track of the current section
493 
494  bracket_pattern = re.compile("\[(.+)\]") # section pattern matching
495 
496  current_section = self._root_section
497  current_section_invalid = False
498 
499  # --- Parse each line...
500  token_stack = []
501 
502  for line in fd.readlines():
503  line = line.strip()
504 
505  # --- Ignore blank lines and comments
506  if len(line) != 0 and not line.startswith('#'):
507 
508  # --- Split line into (name,value) pairs or sections
509 
510  is_continued = '\\' in line # continuation character found
511 
512  token_list = re.split('[#]', line, maxsplit=1) # strip comment
513  token_list = re.split('[=]', token_list[0], maxsplit=1) # parse '='
514 
515  # --- Locate section declaration if any in the line
516  section_match = bracket_pattern.search(token_list[0])
517  if not section_match is None and len(token_stack) == 0:
518 
519  # A definition for a section was found
520  section_name = self._format_section_name(section_match.group(1).strip())
521 
522  # Adding section
523  self._next_section_rank += 1
524  section = self._add_section(section_name, self._next_section_rank)
525  current_section_invalid = (section is None)
526  if not section is None:
527  current_section = section
528  else:
529  if is_continued:
530  if len(token_list) == 2:
531  # Remove continuation character if any
532  if '\\' in token_list[1]:
533  token_list[1] = token_list[1].strip().replace('\\', "", 1)
534  token_stack.append(token_list)
535  continue
536  elif len(token_list) == 1:
537  # partial line, without the '=' separator
538  token_list = self._join_line_tokens(token_list, token_stack)
539  token_stack.append(token_list)
540  continue
541  else:
542  if len(token_stack) > 0:
543  if len(token_list) == 1:
544  # partial line, without the '=' separator
545  token_list = self._join_line_tokens(token_list, token_stack)
546  token_stack = []
547 
548  # --- Adding (key,value) pairs
549  if not current_section_invalid:
550  # A valid [key,pair,comment] definition
551  if len(token_list) == 2:
552  current_section.add_key_value(token_list[0].strip(),
553  sch.expand_env_vars(token_list[1]).strip())
554  #print 'adding (key,value) pair ({0}, {1})'.format(token_list[0].strip(), sch.expand_env_vars(token_list[1]).strip())
555  elif len(token_list) == 1:
556  raise SConfig.ValueNotFound(token_list[0], current_section.name)
557  else:
558  raise SConfig.InvalidSection(current_section.name)
559  #print("SConfig *** ERROR ***: section [{0}] is invalid => corresponding data not recorded".format(current_section.name))

376 
377  if branch is not None:
378 
379  branch.strip("/") # strip first "/" just in case...
380  branch_dirs = branch.split("/") # extract branch dir components
381

383  job_result_dico = self._job_result_dico

899  def get_file_components_from_path(self, file_path):
900  """!
901  Return the list of file components of a file absolute path: file type, image number
902  and possibly epoch index.
903  @code
904  - If the file is of the form /<path>/<file_type>-<image_no>-<epoch>.<extension> a list
905  [file_type, img_no, epoch] is returned
906  - If the file is of the form /<path>/<file_type>-<image_no>.<extension>, a list
907  [file_type, img_no, None] is returned
908  @endcode
909  @param file_path absolute file path of a file (image, catalog, etc.)
910  @code
911  Examples:
912  - get_file_compoments_from_path("starfield-000.1.fits") => [starfield_image.fits, 0, 1]
913  - get_file_compoments_from_path("star_catalog-000.txt") => [star_catalog.txt, 0, None]
914  @endcode
915  @return
916  @see get_file_type_from_path(), get_file_epoch_from_path()
917  """
918 
919  _, filename = os.path.split(file_path)
920  _, fileext = os.path.splitext(filename)
921 
922  regex_dash = re.compile("(.+)\-(.*[0-9])\-(.*[0-9])\.")
923  result = regex_dash.search(filename)
924  if not result is None:
925  return [result.group(1)+fileext, int(result.group(2)), int(result.group(3))]
926  else:
927  regex_dot = re.compile("(.+)\-(.*[0-9])\.")
928  result = regex_dot.search(filename)
929  return [result.group(1)+fileext, int(result.group(2)), None]
